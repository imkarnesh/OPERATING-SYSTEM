#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, size, dir, i, j, total = 0;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    int req[n];
    printf("Enter request sequence:\n");
    for (i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &size);

    printf("Enter direction (0=left, 1=right): ");
    scanf("%d", &dir);

    for (i = 0; i < n - 1; i++)
        for (j = i + 1; j < n; j++)
            if (req[i] > req[j]) {
                int t = req[i];
                req[i] = req[j];
                req[j] = t;
            }

    int current = head;

    if (dir == 1) {
        for (i = 0; i < n; i++)
            if (req[i] >= head) {
                total += abs(current - req[i]);
                current = req[i];
            }
        total += abs(current - (size - 1));
        current = size - 1;

        for (i = n - 1; i >= 0; i--)
            if (req[i] < head) {
                total += abs(current - req[i]);
                current = req[i];
            }
    } else {
        for (i = n - 1; i >= 0; i--)
            if (req[i] <= head) {
                total += abs(current - req[i]);
                current = req[i];
            }
        total += abs(current);
        current = 0;

        for (i = 0; i < n; i++)
            if (req[i] > head) {
                total += abs(current - req[i]);
                current = req[i];
            }
    }

    printf("Total head movement = %d\n", total);
    return 0;
}
