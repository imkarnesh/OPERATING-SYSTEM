#include <stdio.h>
#include <unistd.h>
#include <fcntl.h>

int main() {
    int sourceFile, destinationFile;
    char buffer[100];
    int bytesRead;

    sourceFile = open("source.txt", O_RDONLY);
    destinationFile = open("dest.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    bytesRead = read(sourceFile, buffer, 100);
    while (bytesRead > 0) {
        write(destinationFile, buffer, bytesRead);
        bytesRead = read(sourceFile, buffer, 100);
    }

    close(sourceFile);
    close(destinationFile);
    return 0;
}
