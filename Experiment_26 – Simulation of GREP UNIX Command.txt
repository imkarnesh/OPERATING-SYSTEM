#include <stdio.h>
#include <string.h>

int main() {
    char filename[50], search[50], line[200];
    FILE *fp;

    printf("Enter file name: ");
    scanf("%s", filename);

    printf("Enter word to search: ");
    scanf("%s", search);

    fp = fopen(filename, "r");
    if (fp == NULL) {
        printf("File not found!\n");
        return 1;
    }

    printf("\nMatching Lines:\n");

    while (fgets(line, sizeof(line), fp)) {
        if (strstr(line, search)) {
            printf("%s", line);
        }
    }

    fclose(fp);
    return 0;
}
