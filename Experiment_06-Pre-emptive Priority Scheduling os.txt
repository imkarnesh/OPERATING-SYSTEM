#include <stdio.h>

int main(){
    int n,i,j,t,bt[10],pr[10],wt[10],tat[10];
    float awt=0, atat=0;

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++){
        printf("Enter BT P%d: ",i+1);
        scanf("%d",&bt[i]);
        printf("Enter PR P%d: ",i+1);
        scanf("%d",&pr[i]);
    }

    for(i=0;i<n-1;i++)
        for(j=i+1;j<n;j++)
            if(pr[i]>pr[j]){
                t=pr[i]; pr[i]=pr[j]; pr[j]=t;
                t=bt[i]; bt[i]=bt[j]; bt[j]=t;
            }

    wt[0]=0;
    for(i=1;i<n;i++)
        wt[i]=wt[i-1]+bt[i-1];

    for(i=0;i<n;i++){
        tat[i]=wt[i]+bt[i];
        awt+=wt[i]; atat+=tat[i];
    }

    printf("\nPR BT WT TAT\n");
    for(i=0;i<n;i++)
        printf("%d %d %d %d\n",pr[i],bt[i],wt[i],tat[i]);

    printf("Avg WT=%.2f\nAvg TAT=%.2f",awt/n,atat/n);
    return 0;
}
